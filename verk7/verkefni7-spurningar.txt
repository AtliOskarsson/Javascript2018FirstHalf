1. Callback er breyta sem er keyrð eftir að önnur 
   breyta hefur ferið keyrð og notar það fall til að útvega svar.

function greeting(name) {
    alert('Hello ' + name);
}

function userInput(callback) {
   var name = prompt('Please enter your name.');
   callback(name);
}
// Kallar á userInput með inputið sem greeting
userInput(greeting);
